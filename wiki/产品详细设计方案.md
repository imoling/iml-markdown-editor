### 1.5 Skill 引擎核心逻辑

```typescript
// 类型定义
interface Skill {
  id: string;
  name: string;
  description: string;
  version: string;
  author: string;
  icon: string;
  tags: string[];
  triggers: string[];
  category: string;
  model: ModelConfig;
  variables: Variable[];
  prompts: {
    system: string;
    user: string;
    examples?: string;
  };
  context: ContextConfig;
  output: OutputConfig;
}

interface Variable {
  name: string;
  label: string;
  type: 'text' | 'select' | 'textarea' | 'boolean';
  required: boolean;
  default?: string;
  options?: string[];
  placeholder?: string;
}

interface SkillExecutionContext {
  vibe: string;
  variables: Record<string, any>;
  currentDocContent?: string;
  selectedText?: string;
}

// 核心流程
class SkillEngine {
  // 1. 加载 Skill
  loadSkills(): Skill[]

  // 2. 搜索 Skill（支持模糊匹配）
  searchSkills(query: string): Skill[]

  // 3. 执行 Skill
  async execute(skill: Skill, context: SkillExecutionContext): Promise<string>

  // 4. 解析变量
  parseVariables(template: string, variables: Record<string, any>): string

  // 5. 构建 Prompt
  buildPrompt(skill: Skill, context: SkillExecutionContext): Message[]
}
```

```typescript
// themeStore.ts
interface ThemeState {
  mode: 'light' | 'dark' | 'system';
  effectiveMode: 'light' | 'dark';
}

const useThemeStore = create<ThemeState>((set, get) => ({
  mode: 'system',
  effectiveMode: 'light',

  setMode: (mode: 'light' | 'dark' | 'system') => {
    const effectiveMode = mode === 'system'
      ? window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light'
      : mode;

    document.documentElement.setAttribute('data-theme', effectiveMode);
    set({ mode, effectiveMode });
  },

  toggle: () => {
    const current = get().effectiveMode;
    get().setMode(current === 'light' ? 'dark' : 'light');
  }
}));
```

### 3.4 模糊搜索算法

```typescript
function fuzzyMatch(query: string, target: string): boolean {
  const queryLower = query.toLowerCase().replace(/^\//, '');
  const targetLower = target.toLowerCase();

  let queryIndex = 0;
  for (let i = 0; i < targetLower.length && queryIndex < queryLower.length; i++) {
    if (targetLower[i] === queryLower[queryIndex]) {
      queryIndex++;
    }
  }

  return queryIndex === queryLower.length;
}

function scoreMatch(query: string, target: string): number {
  // 精确前缀匹配 > 单词开头匹配 > 字符连续匹配 > 字符分散匹配
  if (target.startsWith(query)) return 100;
  if (target.split(' ').some(w => w.startsWith(query))) return 80;
  if (target.includes(query)) return 60;
  if (fuzzyMatch(query, target)) return 40;
  return 0;
}
```

### 4.4 冲突检测

```typescript
function detectConflict(newShortcut: string, existingBindings: ShortcutBinding[]): ShortcutBinding | null {
  for (const binding of existingBindings) {
    if (binding.mac === newShortcut || binding.win === newShortcut) {
      return binding;
    }
  }
  return null;
}
```

```typescript
    set(key: string, value: any): void;
    remove(key: string): void;
  };

  // UI
  ui: {
    showNotification(message: string, type: 'info' | 'success' | 'warning' | 'error'): void;
    openPanel(panelId: string): void;
    closePanel(panelId: string): void;
  };

  // 工具
  utils: {
    showInputBox(options: InputBoxOptions): Promise<string>;
    showQuickPick<T>(items: QuickPickItem<T>[]): Promise<T | undefined>;
  };
}

// 插件入口函数
export function activate(api: IMLPluginAPI) {
  // 注册命令
  api.commands.register({
    id: 'my-plugin.hello',
    label: 'Hello World',
    execute: () => {
      api.ui.showNotification('Hello from plugin!', 'info');
    }
  });

  // 返回清理函数
  return () => {
    // 插件禁用时清理资源
  };
}
```
